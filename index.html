function closeModal(event) { // Проверяем, был ли клик по самому overlay (не по modal) if (event && event.target !== document.getElementById('modalOverlay')) { return; } const modal = document.getElementById('modal'); const overlay = document.getElementById('modalOverlay'); // Добавляем класс для анимации свайпа modal.classList.add('closing'); // Ждем завершения анимации, затем закрываем setTimeout(() => { modal.classList.remove('closing'); overlay.classList.remove('open'); }, 350); // соответствует длительности анимации (0.35s) }
